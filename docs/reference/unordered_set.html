cstl_unordered_set

Hash-based collection of unique elements with O(1) average time complexity. Equivalent to std::unordered_set<T>.

UnorderedHash Tablehashtable.h

without USE_CSTL_FUNC
USET(int) s = cstl_unordered_set(int);
with USE_CSTL_FUNC
#define USE_CSTL_FUNC
USET(int) s = new_unordered_set(int);

Example

c
#define USE_CSTL_FUNC
#include "opencstl.h"

void example_uset(void) {
    USET(int) s = new_unordered_set(int);

    insert(s, 10);
    insert(s, 20);
    insert(s, 10); // duplicate, ignored

    if (find(s, 20)) {
        printf("20 is in the set\n");
    }

    destroy(s);
}
