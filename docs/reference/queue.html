cstl_queue

FIFO (First-In First-Out) container adapter. Equivalent to std::queue<T>.

Adapterqueue.h

without USE_CSTL_FUNC
QUEUE(int) q = cstl_queue(int);
with USE_CSTL_FUNC
#define USE_CSTL_FUNC
QUEUE(int) q = new_queue(int);

Example

c
#define USE_CSTL_FUNC
#include "opencstl.h"

void example_queue(void) {
    QUEUE(int) q = new_queue(int);

    push(q, 1);
    push(q, 2);

    printf("Front: %d\n", front(q)); // 1
    pop(q);
    printf("Front after pop: %d\n", front(q)); // 2

    destroy(q);
}
