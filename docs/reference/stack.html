cstl_stack

LIFO (Last-In First-Out) container adapter. Equivalent to std::stack<T>.

Adapterstack.h

without USE_CSTL_FUNC
STACK(int) s = cstl_stack(int);
with USE_CSTL_FUNC
#define USE_CSTL_FUNC
STACK(int) s = new_stack(int);

Example

c
#define USE_CSTL_FUNC
#include "opencstl.h"

void example_stack(void) {
    STACK(int) s = new_stack(int);

    push(s, 100);
    push(s, 200);

    printf("Top: %d\n", top(s)); // 200
    pop(s);
    printf("Top after pop: %d\n", top(s)); // 100

    destroy(s);
}
