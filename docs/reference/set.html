cstl_set

Sorted collection of unique elements backed by a Red-Black tree. Equivalent to std::set<T>.

AssociativeSortedBidirectional Iteratortree.h

Declaration

without USE_CSTL_FUNC
SET(int) s = cstl_set(int, COMPARE(int));
with USE_CSTL_FUNC
#define USE_CSTL_FUNC
SET(int) s = new_set(int, COMPARE(int));

Example

c
#define USE_CSTL_FUNC
#include "opencstl.h"

void example_set(void) {
    SET(int) s = new_set(int, COMPARE(int));

    insert(s, 30);
    insert(s, 10);
    insert(s, 20);
    insert(s, 10); // duplicate, ignored

    // sorted traversal: 10, 20, 30
    for (int *it = begin(s); it != end(s); it = next(it)) {
        printf("%d ", *it);
    }

    destroy(s);
}
