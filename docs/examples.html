USE_CSTL_FUNC — Short Names

Define USE_CSTL_FUNC before the include to enable short names identical to C++ STL. Without it, use the full cstl_ prefixed names.

c
// ─── without USE_CSTL_FUNC ───────────────────────────────────────────
#include "opencstl.h"

VECTOR(int) v = cstl_vector(int);
cstl_push_back(v, 42);
cstl_free(v);

// ─── with USE_CSTL_FUNC ──────────────────────────────────────────────
#define USE_CSTL_FUNC
#include "opencstl.h"

VECTOR(int) v = new_vector(int);
push_back(v, 42);
destroy(v);
ℹ

Short name mappings: new_vector→cstl_vector  |  push_back→cstl_push_back  |  destroy→cstl_free  |  size→cstl_size  |  begin→cstl_begin  |  next→cstl_next   and so on.

Vector

Dynamic array with O(1) amortized append and iterator loop. v[i] subscript access works just like a plain C array.

c
#define USE_CSTL_FUNC
#include "opencstl.h"

void example_vector(void) {
    VECTOR(int) arr = new_vector(int);

    for (int i = 0; i < 10; i++) {
        push_back(arr, i);
    }
    for (int *it = begin(arr); it != end(arr); it++) {
        printf("[%3d]", *it);
    }
    puts("");
    destroy(arr);
}
✓

Iterator arithmetic (it++) works on vector — elements are contiguous in memory, just like a raw array.

List

Doubly-linked list with O(1) insert and erase at any position. Must use next(it) to advance.

c
void example_list(void) {
    LIST(int) arr = new_list(int);

    for (int i = 10; i >= 0; i--) {
        push_back(arr, i);
    }
    for (int *it = begin(arr); it != end(arr); it = next(it)) {
        printf("[%3d]", *it);
    }
    puts("");
    destroy(arr);
}
⚠

List iterators are node pointers. Always use next(it) / prev(it) to traverse — never it++.

Set

Sorted set backed by a Red-Black tree. Duplicates are silently discarded. COMPARE(TYPE) auto-selects the built-in comparator for the given type.

c
void example_set(void) {
    SET(int) set = new_set(int, COMPARE(int));

    for (int i = 0; i < 10; i++) {
        insert(set, rand() % 100);
    }
    for (int *it = begin(set); it != end(set); it = next(it)) {
        printf("[%3d]", *it);
    }
    puts("");
    destroy(set);
}

Map

Sorted key-value store. The iterator *it gives the key. Use cstl_value(it, VALUE_TYPE) to access the value.

c
void example_map(void) {
    char *containers[] = {
        "vector", "list", "set", "map", "stack",
        "queue", "priority_queue", "unordered_map", "unordered_set"
    };
    size_t len = sizeof(containers) / sizeof(containers[0]);
    printf("Number of elements: %zu\n", len);

    // new_map(KEY, VALUE, comparator)
    MAP(int) d = new_map(int, char*, COMPARE(int));

    for (int i = 0; i < len; i++) {
        insert(d, i, containers[i]);
    }
    // reverse iteration: *it = key, cstl_value(it, char*) = value
    for (int *it = rbegin(d); it != rend(d); it = prev(it)) {
        printf("[%3d : %s]\n", *it, cstl_value(it, char*));
    }
    puts("");
    destroy(d);
}
💡

cstl_value(it, TYPE) is an O(1) macro that also works as an lvalue — you can write cstl_value(it, int) = 99 to update the value in-place without a re-insert.

Deque

Subscript access, variadic insert (single or N copies), and direct compatibility with standard C qsort.

c
#define USE_CSTL_FUNC
// Defining USE_CSTL_FUNC ensures that short, concise names are used.
#include "opencstl.h"
#include <stdlib.h>

// Comparison function for sorting integers in ascending order.
int cmp_int(const void *a, const void *b) {
    return *((int *) a) - *((int *) b);
}

void example_deque(void) {
    DEQUE(int) q = new_deque(int);

    for (int i = 0; i < 10; i++) {
        int val = rand32() % 100;
        // It uses the same names as the C++ STL.
        push_back(q, val);
    }

    // You can use `insert` in a similar way to overloading in STL.

    // Inserts 777 at the 5th position.
    insert(q, q + 5, 777);

    // Inserts three 999s at the 8th position.
    insert(q, q + 8, 3, 999);

    for (int i = 0; i < size(q); i++) {
        // You can access elements in a deque or vector using [].
        printf("[%3d]", q[i]);
    }
    puts("");

    // You can use existing C functions.
    qsort(q, size(q), sizeof(int), cmp_int);

    for (int i = 0; i < size(q); i++) {
        printf("[%3d]", q[i]);
    }
    puts("");

    // Use `destroy` to free up memory.
    destroy(q);
}

int main(void) {
    example_deque();
    return 0;
}
✓

insert(q, q+5, 777) inserts one value; insert(q, q+8, 3, 999) inserts N copies — same function name, different parameter count, just like C++ STL overloading.

Full Example — example_basic_01.c

All four containers in one complete program. Drop this file into your project and compile directly.

c
#define USE_CSTL_FUNC
#include "opencstl.h"

void example_vector(void) {
    VECTOR(int) arr = new_vector(int);
    for (int i = 0; i < 10; i++) { push_back(arr, i); }
    for (int *it = begin(arr); it != end(arr); it++) { printf("[%3d]", *it); }
    puts("");
    destroy(arr);
}

void example_list(void) {
    LIST(int) arr = new_list(int);
    for (int i = 10; i >= 0; i--) { push_back(arr, i); }
    for (int *it = begin(arr); it != end(arr); it = next(it)) { printf("[%3d]", *it); }
    puts("");
    destroy(arr);
}

void example_set(void) {
    SET(int) set = new_set(int, COMPARE(int));
    for (int i = 0; i < 10; i++) { insert(set, rand() % 100); }
    for (int *it = begin(set); it != end(set); it = next(it)) { printf("[%3d]", *it); }
    puts("");
    destroy(set);
}

void example_map(void) {
    char *containers[] = {
        "vector", "list", "set", "map", "stack",
        "queue", "priority_queue", "unordered_map", "unordered_set"
    };
    size_t len = sizeof(containers) / sizeof(containers[0]);
    printf("Number of elements: %zu\n", len);
    MAP(int) d = new_map(int, char*, COMPARE(int));
    for (int i = 0; i < len; i++) { insert(d, i, containers[i]); }
    for (int *it = rbegin(d); it != rend(d); it = prev(it)) {
        printf("[%3d : %s]\n", *it, cstl_value(it, char*));
    }
    puts("");
    destroy(d);
}

int main(void) {
    example_vector();
    example_list();
    example_set();
    example_map();
    return 0;
}
